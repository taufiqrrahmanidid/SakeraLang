// Hello World
sango nama = "Madura"
toles "Halo dari "
toles nama
// Fibonacci dengan rekursi
fungsi fibonacci(n) {
    mon n < 2 {
        mareh n
    }
    laen {
        sango a = fibonacci(n - 1)
        sango b = fibonacci(n - 2)
        mareh a + b
    }
}

toles fibonacci(10)  // Output: 55
// Functions
fungsi tambah(a, b) {
    mareh a + b
}

sango hasil = tambah(5, 3)
toles hasil  // Output: 8
